CREATE OR REPLACE 
	PROCEDURE UpdateActor 
		(
			p_ID IN ACTOR.ID%TYPE, 
			p_NAME IN ACTOR.NAME%TYPE, 
			p_SURNAME IN ACTOR.SURNAME%TYPE, 
			p_COUNTRY IN ACTOR.COUNTRY%TYPE, 
			p_BDAY IN ACTOR.BDAY%TYPE
		)
	IS
	BEGIN 
		UPDATE ACTOR
			SET 
				NAME = p_NAME, 
				SURNAME = p_SURNAME, 
				COUNTRY = p_COUNTRY, 
				BDAY = p_BDAY
		WHERE ID = p_ID;
	END;
