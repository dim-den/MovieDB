	PROCEDURE GenerateActors 
		(
			rows_count number
		)
	IS
	BEGIN 
		FOR k IN 1..rows_count LOOP 
			ActorPackage.AddActor(DBMS_RANDOM.string('a', 10), DBMS_RANDOM.string('a', 10), DBMS_RANDOM.string('a', 10), add_months(SYSDATE, -12 * floor(dbms_random.value(10, 40))) + DBMS_RANDOM.value(0, 366));
		END LOOP;
	END;
