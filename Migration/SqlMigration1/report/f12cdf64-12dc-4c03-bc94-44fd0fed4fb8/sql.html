CREATE VIEW MOVIE_USER.MGV_ALL_CONNECTIONS (
	PROJECT_ID, 
	PROJECT_NAME, 
	CONNECTION_ID, 
	CONNECTION_NAME, 
	HOST, 
	PORT, 
	USERNAME, 
	DBURL)
AS 
	SELECT 
		MD_PROJECTS.ID AS PROJECT_ID, 
		MD_PROJECTS.PROJECT_NAME AS PROJECT_NAME, 
		MD_CONNECTIONS.ID AS CONNECTION_ID, 
		MD_CONNECTIONS.NAME AS CONNECTION_NAME, 
		MD_CONNECTIONS.HOST AS HOST, 
		MD_CONNECTIONS.PORT AS PORT, 
		MD_CONNECTIONS.USERNAME AS USERNAME, 
		MD_CONNECTIONS.DBURL AS DBURL
	FROM MOVIE_USER.MD_PROJECTS, MOVIE_USER.MD_CONNECTIONS
	WHERE MD_CONNECTIONS.PROJECT_ID_FK = MD_PROJECTS.ID
