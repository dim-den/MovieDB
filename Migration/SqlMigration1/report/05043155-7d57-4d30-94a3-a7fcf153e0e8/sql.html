CREATE OR REPLACE 
	PROCEDURE UpdateFilmGenre 
		(
			p_ID IN FILM_GENRES.ID%TYPE, 
			p_FILM_ID IN FILM_GENRES.FILM_ID%TYPE, 
			p_GENRE_ID IN FILM_GENRES.GENRE_ID%TYPE
		)
	IS
	BEGIN 
		UPDATE FILM_GENRES
			SET FILM_ID = p_FILM_ID, GENRE_ID = p_GENRE_ID
		WHERE ID = p_ID;
	END;
