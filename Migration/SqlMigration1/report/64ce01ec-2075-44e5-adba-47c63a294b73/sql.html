CREATE OR REPLACE 
	PROCEDURE UpdateGenre 
		(
			p_ID IN Genre.ID%TYPE, 
			p_NAME IN Genre.NAME%TYPE
		)
	IS
	BEGIN 
		UPDATE Genre
			SET NAME = p_NAME
		WHERE ID = p_ID;
	END;

	@P_NAME varchar(1020)
