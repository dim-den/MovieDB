	VIEW MGV_ALL_VIEW_TRIGGERS (
		PROJECT_ID, 
		PROJECT_NAME, 
		CONNECTION_ID, 
		HOST, 
		PORT, 
		USERNAME, 
		CATALOG_ID, 
		CATALOG_NAME, 
		DUMMY_FLAG, 
		SCHEMA_ID, 
		SCHEMA_NAME, 
		VIEW_ID, 
		VIEW_NAME, 
		TRIGGER_ID, 
		TRIGGER_NAME)
		AS 
			SELECT 
				md_projects.id project_id, 
				md_projects.project_name project_name, 
				md_connections.id connection_id, 
				md_connections.host host, 
				md_connections.port port, 
				md_connections.username username, 
				md_catalogs.id catalog_id, 
				md_catalogs.catalog_name catalog_name, 
				md_catalogs.dummy_flag dummy_flag, 
				md_schemas.id schema_id, 
				md_schemas.name schema_name, 
				md_views.id view_id, 
				md_views.view_name view_name, 
				md_triggers.id trigger_id, 
				md_triggers.trigger_name trigger_name
			FROM 
				md_projects, 
				md_connections, 
				md_catalogs, 
				md_schemas, 
				md_views, 
				md_triggers
			WHERE 
				md_triggers.table_or_view_id_fk = md_views.id AND 
				md_views.schema_id_fk = md_schemas.id AND 
				md_schemas.catalog_id_fk = md_catalogs.id AND 
				md_catalogs.connection_id_fk = md_connections.id AND 
				md_connections.project_id_fk = md_projects.id;

CREATE VIEW MOVIE_USER.MGV_ALL_VIEW_TRIGGERS (
	PROJECT_ID, 
	PROJECT_NAME, 
	CONNECTION_ID, 
	HOST, 
	PORT, 
	USERNAME, 
	CATALOG_ID, 
	CATALOG_NAME, 
	DUMMY_FLAG, 
	SCHEMA_ID, 
	SCHEMA_NAME, 
	VIEW_ID, 
	VIEW_NAME, 
	TRIGGER_ID, 
	TRIGGER_NAME)
AS 
	SELECT 
		MD_PROJECTS.ID AS PROJECT_ID, 
		MD_PROJECTS.PROJECT_NAME AS PROJECT_NAME, 
		MD_CONNECTIONS.ID AS CONNECTION_ID, 
		MD_CONNECTIONS.HOST AS HOST, 
		MD_CONNECTIONS.PORT AS PORT, 
		MD_CONNECTIONS.USERNAME AS USERNAME, 
		MD_CATALOGS.ID AS CATALOG_ID, 
		MD_CATALOGS.CATALOG_NAME AS CATALOG_NAME, 
		MD_CATALOGS.DUMMY_FLAG AS DUMMY_FLAG, 
		MD_SCHEMAS.ID AS SCHEMA_ID, 
		MD_SCHEMAS.NAME AS SCHEMA_NAME, 
		MD_VIEWS.ID AS VIEW_ID, 
		MD_VIEWS.VIEW_NAME AS VIEW_NAME, 
		MD_TRIGGERS.ID AS TRIGGER_ID, 
		MD_TRIGGERS.TRIGGER_NAME AS TRIGGER_NAME
	FROM 
		MOVIE_USER.MD_PROJECTS, 
		MOVIE_USER.MD_CONNECTIONS, 
		MOVIE_USER.MD_CATALOGS, 
		MOVIE_USER.MD_SCHEMAS, 
		MOVIE_USER.MD_VIEWS, 
		MOVIE_USER.MD_TRIGGERS
