CREATE OR REPLACE 
	FUNCTION GetFilmAvgScore 
		(
			p_FILM_ID FILM_REVIEW.FILM_ID%TYPE
		)
		 RETURN float
	IS
	
		avg_score float;
	BEGIN 

		SELECT avg(score)
		INTO avg_score
		FROM FILM_REVIEW
		WHERE film_id = p_FILM_ID;

		RETURN avg_score;

	END;

CREATE FUNCTION MOVIE_USER.FILMREVIEWPACKAGE$GETFILMAVGSCORE 
( 
	@P_FILM_ID numeric(19, 0)
)
RETURNS float(53)
AS 
	BEGIN

		EXECUTE ssma_oracle.db_fn_check_init_package 'MOVIE_USER', 'FILMREVIEWPACKAGE'

		DECLARE
			@AVG_SCORE float(53)

		SELECT @AVG_SCORE = avg(FILM_REVIEW.SCORE)
		FROM MOVIE_USER.FILM_REVIEW
		WHERE FILM_REVIEW.FILM_ID = @P_FILM_ID

		RETURN @AVG_SCORE

	END
